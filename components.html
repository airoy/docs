# 组件开发规范

# base-table

# 使用案例

<BaseTable :columns="columns" :url="url" :isDragable="true" :formData="formData" @on-end="end">
  <!--内部为默认插槽，可以写入搜索条件-->
  <Row>
    <i-col span="16">
      <!--表单组 form-group-->
      <div class="form-group">
        <!--表单 form-inline-->
        <Input class="form-inline" placeholder="请输入有效关键字查询"/>
      </div>
    </i-col>
    <i-col span="8">
      <Button type="success" icon="ios-search" @click="search">查询</Button>
      <Button type="warning" icon="ios-refresh">重置</Button>
      <Button type="info" icon="ios-add-circle-outline">新增</Button>
    </i-col>
  </Row>
</BaseTable>

//  rows: 10, page: 1 为默认，无需再次传入
search() {
  this.formData = {
    sidx: 1,
    sord: 'desc'
  }
}

# API

# props

属性 说明 类型 默认值
columns 表格列的配置描述 Array []
highlight-row 是否支持高亮选中的行，即单选 Boolean false
formData 搜索条件查询（查询时只需改变条件即可自动查询） Object {}
url 查询的接口地址（初始化时赋值） String
isDragable 是否支持拖拽 Boolean false

# events

事件名 说明 返回值
on-start 拖拽开始 被拖拽的元素索引
on-end 拖拽结束 Object（包含from: 拖拽开始索引，to: 拖拽结束索引，value: 拖拽元素row对象）
on-current-change 开启 highlight-row 后有效，当表格的当前行发生变化的时候会触发 currentRow：当前高亮行的数据 oldCurrentRow：上一次高亮的数据
on-selection-change 在多选模式下有效，只要选中项发生变化时就会触发 selection：已选项数据

# base-modal

# 使用案例

<!--父组件-->
<!--动态给modalType赋值（child.vue）加载组件， type为组件的展示状态（增，改，查等等）-->
<div v-if="modalType">
  <component :is="modalType" :type="type"></component>
</div>
<!--子组件-->
<!--child.vue-->
<!--v-bind="$attrs" 固定写法 title为模态标题； 默认slot为内容插槽；slot="footer" 为底部插槽，一般加入按钮-->
<BaseModal v-bind="$attrs" :title="title">
 <!--弹框主体内容-->
  <div>
    <p>我是内容</p>
  </div>
  <!--自定义模态窗的按钮-->
  <div slot="footer">
    <Button type="default" @click="cancel">关闭</Button>
    <!--type为增，删，改，查，编辑，审核等-->
    <template v-if="type === 1">
        <Button type="warning">不通过</Button>
        <Button type="info">通过</Button>
    </template>
  </div>
</BaseModal>

注意

  • 子组件要有关闭方法，调用父组件的方法
// 父组件
cancel() {
  this.modalType = null
}
// 子组件
cancel() {
  this.$parent.cancel()
}
  • 子组有监听器
export default {
    name: '',
    inheritAttrs: false,
    data () {
        return {
            title: '',
            type: -1
        }
    },
    methods: {
         // 固定写法
        cancel() {
            this.$parent.cancel()
        },
        // 不同type的所有逻辑写在这
        _diffType(val) {
            this.type = val
            if (val === 0) {
                this.title = '查看'
            } else if (val === 1) {
                this.title = '审核'
            }
        }
    },
    // 监听type调用自定义方法
    watch: {
        '$attrs.type': {
            handler: function(val) {
                this._diffType(val)
            },
            deep: true,
            immediate: true
        }
    }
}

# base-upload

<BaseUpload @successUpload="handlerSuccessUpload"></BaseUpload>

# API

# props

属性 说明 类型 默认值
multiple 是否支持多张上传 Boolean false

# events

事件名 说明 返回值
successUpload 图片上传成功回调 multiple: true 为多张上传 ，返回数组; multiple: false返回当前图片信息，object

# base-editor

<BaseEditor ref="editor" @input="setContent"></BaseEditor>

# props

属性 说明 类型 默认值
value 绑定的值，可使用 v-model 双向绑定 String Number
id 当前编辑器的id String 随机字符串（非必填）
toolbar 编辑器工具 Array [...]

# API

# events

事件名 说明 返回值
input 编辑器返回的内容（实时） String
setContent 设置编辑器内容 (this.$refs.editor.setContent(val))
getContent 获取编辑器内容 String (this.$refs.editor.getContent)
